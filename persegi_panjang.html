import java.util.Scanner;

    public class PersegiPanjang {
     /**
    @param args Argumen baris perintah (tidak digunakan).
     */
  
    public static void main(String[] args) {
        
        Scanner scanner = new Scanner(System.in);
        System.out.print("Masukkan panjang persegi panjang: ");
        double panjang = scanner.nextDouble();

        System.out.print("Masukkan lebar persegi panjang: ");
        double lebar = scanner.nextDouble();

 
        double area = calculateArea(panjang, lebar);
        System.out.println("Luas persegi panjang adalah: " + area);

        scanner.close();
    }


   /**
     * @param panjang Panjang persegi panjang.
     * @param lebar  Lebar persegi panjang.
     * @return Luas dari persegi panjang.
     */

    public static double calculateArea(double panjang, double lebar) {
        return panjang * lebar;
    }
}
